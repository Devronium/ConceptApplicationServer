Example: 
import standard.arch.zip



class Main {

	function Main() {

		var archive="test.zip";

		if (!(var za=zip_open(archive, ZIP_CREATE, var err))) {

			echo "Error: $err\n";

			return 1;

		}



		var zs=zip_source_buffer(za, "Hello world ... from a ZIP file !");



		zip_add(za, "teststring.txt",zs);

		zip_close(za);

	}

}
